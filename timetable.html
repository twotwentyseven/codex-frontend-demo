    <codex-timetable-alt
		:disable-unavailable-events="false"
		start-date="2022-10-01"
		:show-bookmark-toggle="true"
		:show-events-when-logged-out="true"
		event-detail-url="event.html"
		:show-date-on-event-card="true"
		:days-to-show="{
			640: 1,
			1024: 2,
			1200: 3,
		}"
		:nav-carousel-settings="{
			slidesPerView: 'auto',
			threshold: 10,
		}"
		:filters="{
			location: {
				selector: 'studio.location.id',
				options: 'api/locations:id:name',
				title: $t('timetable.location')
			},
			instructor: {
				selector: 'instructor.id',
				options: 'api/instructors:id:first_name',
				title: $t('timetable.instructors')
			},
			workoutType: {
				selector: 'event_type.id',
				options: 'api/event_types:id:name',
				title: $t('event.workout_type'),
				multi: true
			},
			
		}">

		<template v-slot:event-modal="{ id, eventObject, purchaseUrl, isModal, closeModal }">

			<codex-event :id="id" :event-object="eventObject" account-url="account.html" purchase-url="purchase.html" :show-when-logged-out="true" :is-modal="isModal" :close-modal="closeModal">

			</codex-event>

		</template>
	</codex-timetable-alt>
